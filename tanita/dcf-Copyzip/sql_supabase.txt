-- Criar a tabela
CREATE TABLE IF NOT EXISTS user_preferences (
    id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
    user_id TEXT UNIQUE NOT NULL,
    active_investment JSONB,
    missions_state JSONB,
    bonus_claimed BOOLEAN DEFAULT FALSE,
    notifications JSONB,
    created_at TIMESTAMPTZ DEFAULT NOW(),
    updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Ativar seguranca por linha
ALTER TABLE user_preferences ENABLE ROW LEVEL SECURITY;

-- Policies de seguranca
CREATE POLICY "Users can read own preferences" ON user_preferences FOR SELECT USING (auth.uid()::text = user_id);
CREATE POLICY "Users can insert own preferences" ON user_preferences FOR INSERT WITH CHECK (auth.uid()::text = user_id);
CREATE POLICY "Users can update own preferences" ON user_preferences FOR UPDATE USING (auth.uid()::text = user_id);

-- Funcao para actualizar data
CREATE OR REPLACE FUNCTION update_updated_at() RETURNS TRIGGER AS $$ BEGIN NEW.updated_at = NOW(); RETURN NEW; END; $$ LANGUAGE plpgsql;

-- Trigger
DROP TRIGGER IF EXISTS update_user_preferences_updated_at ON user_preferences;
CREATE TRIGGER update_user_preferences_updated_at BEFORE UPDATE ON user_preferences FOR EACH ROW EXECUTE FUNCTION update_updated_at();
-- ========================================================
-- ONLINE USERS — allow admin (authenticated role) to read
-- Run in Supabase SQL Editor if admin notifications to
-- "online users" shows 0 recipients.
-- ========================================================
ALTER TABLE online_users ENABLE ROW LEVEL SECURITY;

-- Users can write/update their own row
CREATE POLICY "Users update own online status" ON online_users
    FOR ALL USING (auth.uid() = user_id);

-- Any authenticated user (admin included) can read all rows
CREATE POLICY "Authenticated users read online_users" ON online_users
    FOR SELECT USING (auth.role() = 'authenticated');
